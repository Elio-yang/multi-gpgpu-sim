1 /*
2  * DSARP.h
3  *
4  * This a re-implementation of the refresh mechanisms proposed in Chang et al.,
5  * "Improving DRAM Performance by Parallelizing Refreshes with Accesses", HPCA
6  * 2014.
7  *
8  * Note: the re-implementation of DSARP has not been widely tested across
9  * different benchmarks and parameters. However, timing violations of
10  * SARP/DSARP have been checked.
11  *
12  * Usage: The "type" determines the refresh mechanisms.
13  * Examples:
14  * DSARP::Org test_org = DSARP::Org::DSARP_8Gb_x8;
15  *
16  * DSARP* dsddr3_ab = new DSARP(test_org,
17  * DSARP::Speed::DSARP_1333, DSARP::Type::REFAB, 8);
18  *
19  * DSARP* dsddr3_pb = new DSARP(test_org,
20  * DSARP::Speed::DSARP_1333, DSARP::Type::REFPB, 8);
21  *
22  * DSARP* dsddr3_darp = new DSARP(test_org,
23  * DSARP::Speed::DSARP_1333, DSARP::Type::DARP, 8);
24  *
25  * DSARP* dsddr3_sarp = new DSARP(test_org,
26  * DSARP::Speed::DSARP_1333, DSARP::Type::SARP, 8);
27  *
28  * DSARP* dsddr3_dsarp = new DSARP(test_org,
29  * DSARP::Speed::DSARP_1333, DSARP::Type::DSARP, 8);
30  *
31  * Created on: Mar 16, 2015
32  * Author: kevincha
33  */
