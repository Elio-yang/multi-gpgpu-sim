52 template<unsigned BSIZE> class mem_storage {
53 public:
54  mem_storage( const mem_storage &another )
55  {
56  m_data = (unsigned char*)calloc(1,BSIZE);
57  memcpy(m_data,another.m_data,BSIZE);
58  }
59  mem_storage()
60  {
61  m_data = (unsigned char*)calloc(1,BSIZE);
62  }
63  ~mem_storage()
64  {
65  free(m_data);
66  }
67 
68  void write( unsigned offset, size_t length, const unsigned char *data )
69  {
70  assert( offset + length <= BSIZE );
71  memcpy(m_data+offset,data,length);
72  }
73 
74  void read( unsigned offset, size_t length, unsigned char *data ) const
75  {
76  assert( offset + length <= BSIZE );
77  memcpy(data,m_data+offset,length);
78  }
79 
80  void print( const char *format, FILE *fout ) const
81  {
82  unsigned int *i_data = (unsigned int*)m_data;
83  for (int d = 0; d < (BSIZE / sizeof(unsigned int)); d++) {
84  if (d % 8 == 0) {
85  fprintf(fout, "\n");
86  }
87  fprintf(fout, format, i_data[d]);
88  fprintf(fout, " ");
89  }
90  fprintf(fout, "\n");
91  fflush(fout);
92  }
93 
94 private:
95  unsigned m_nbytes;
96  unsigned char *m_data;
97 };
