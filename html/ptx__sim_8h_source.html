76  ptx_reg_t(unsigned x)
77  {
78  bits.ms = 0;
79  bits.ls = 0;
80  u128.low=0;
81  u128.lowest=0;
82  u128.highest=0;
83  u128.high=0;
84  s8=0;

87  s64=0;
88  u8=0;
89  u16=0;
90  u64=0;
91  f16=0;
92  f32=0;
93  f64=0;
94  pred=0;
95  u32 = x;
96  }
97  operator unsigned int() { return u32;}
98  operator unsigned short() { return u16;}
99  operator unsigned char() { return u8;}
100  operator unsigned long long() { return u64;}
101 
102  void mask_and( unsigned ms, unsigned ls )
103  {
104  bits.ms &= ms;
105  bits.ls &= ls;
106  }
107 
108  void mask_or( unsigned ms, unsigned ls )
109  {
110  bits.ms |= ms;
111  bits.ls |= ls;
112  }
113  int get_bit( unsigned bit )
114  {
115  if ( bit < 32 )
116  return(bits.ls >> bit) & 1;
117  else
118  return(bits.ms >> (bit-32)) & 1;
119  }

122  signed short s16;
123  signed int s32;
124  signed long long s64;

236  void set_target( const char *sm_ver, const char *ext, const char *ext2 )
237  {
238  assert( m_valid );
239  m_sm_version_str = sm_ver;
240  check_target_extension(ext);
241  check_target_extension(ext2);
242  sscanf(sm_ver,"%u",&m_sm_version);
243  m_sm_version_valid=true;
244  }
245  float ver() const { assert(m_valid); return m_ptx_version; }
246  unsigned target() const { assert(m_valid&&m_sm_version_valid); return m_sm_version; }
247  unsigned extensions() const { assert(m_valid); return m_ptx_extensions; }
248 private:
249  void check_target_extension( const char *ext )
250  {
251  if( ext ) {
252  if( !strcmp(ext,"texmode_independent") )
253  m_texmode_unified=false;
254  else if( !strcmp(ext,"texmode_unified") )
255  m_texmode_unified=true;
256  else if( !strcmp(ext,"map_f64_to_f32") )
257  m_map_f64_to_f32 = true;
258  else abort();
259  }
260  }
261 
262  bool m_valid;
263  float m_ptx_version;
264  unsigned m_sm_version_valid;

266  bool m_texmode_unified;
