167  {
168  assert(gpgpu_dram_timing_opt);
169  if (strchr(gpgpu_dram_timing_opt, '=') == NULL) {
170  // dram timing option in ordered variables (legacy)
171  // Disabling bank groups if their values are not specified
172  nbkgrp = 1;

176  &nbk,&tCCD,&tRRD,&tRCD,&tRAS,&tRP,&tRC,&CL,&WL,&tCDLR,&tWR,&nbkgrp,&tCCDL,&tRTPL);
177  } else {
178  // named dram timing options (unordered)
179  option_parser_t dram_opp = option_parser_create();
180 
181  option_parser_register(dram_opp, "nbk", OPT_UINT32, &nbk, "number of banks", "");
182  option_parser_register(dram_opp, "CCD", OPT_UINT32, &tCCD, "column to column delay", "");
183  option_parser_register(dram_opp, "RRD", OPT_UINT32, &tRRD, "minimal delay between activation of rows in different banks", "");
184  option_parser_register(dram_opp, "RCD", OPT_UINT32, &tRCD, "row to column delay", "");
185  option_parser_register(dram_opp, "RAS", OPT_UINT32, &tRAS, "time needed to activate row", "");
186  option_parser_register(dram_opp, "RP", OPT_UINT32, &tRP, "time needed to precharge (deactivate) row", "");
187  option_parser_register(dram_opp, "RC", OPT_UINT32, &tRC, "row cycle time", "");
188  option_parser_register(dram_opp, "CDLR", OPT_UINT32, &tCDLR, "switching from write to read (changes tWTR)", "");
189  option_parser_register(dram_opp, "WR", OPT_UINT32, &tWR, "last data-in to row precharge", "");
190 
191  option_parser_register(dram_opp, "CL", OPT_UINT32, &CL, "CAS latency", "");
192  option_parser_register(dram_opp, "WL", OPT_UINT32, &WL, "Write latency", "");
193 
194  //Disabling bank groups if their values are not specified
195  option_parser_register(dram_opp, "nbkgrp", OPT_UINT32, &nbkgrp, "number of bank groups", "1");
196  option_parser_register(dram_opp, "CCDL", OPT_UINT32, &tCCDL, "column to column delay between accesses to different bank groups", "0");
197  option_parser_register(dram_opp, "RTPL", OPT_UINT32, &tRTPL, "read to precharge delay between accesses to different bank groups", "0");
198 
199  option_parser_delimited_string(dram_opp, gpgpu_dram_timing_opt, "=:;");
200  fprintf(stdout, "DRAM Timing Options:\n");
201  option_parser_print(dram_opp, stdout);
202  option_parser_destroy(dram_opp);
203  }
204 
205  int nbkt = nbk/nbkgrp;
206  unsigned i;
207  for (i=0; nbkt>0; i++) {
208  nbkt = nbkt>>1;
209  }
210  bk_tag_length = i;
211  assert(nbkgrp>0 && "Number of bank groups cannot be zero");
212  tRCDWR = tRCD-(WL+1);
213  tRTW = (CL+(BL/data_command_freq_ratio)+2-WL);
214  tWTR = (WL+(BL/data_command_freq_ratio)+tCDLR);
215  tWTP = (WL+(BL/data_command_freq_ratio)+tWR);
216  dram_atom_size = BL * busW * gpu_n_mem_per_ctrlr; // burst length x bus width x # chips per partition
217 
218  assert( m_n_sub_partition_per_memory_channel > 0 );
219  assert( (nbk % m_n_sub_partition_per_memory_channel == 0)
220  && "Number of DRAM banks must be a perfect multiple of memory sub partition");
221  m_n_mem_sub_partition = m_n_mem * m_n_sub_partition_per_memory_channel;
222  fprintf(stdout, "Total number of memory sub partition = %u\n", m_n_mem_sub_partition);
223 
224  m_address_mapping.init(m_n_mem, m_n_sub_partition_per_memory_channel);
225  m_L2_config.init(&m_address_mapping);

227  m_valid = true;
228  icnt_flit_size = 32; // Default 32
