111  // Inject refresh at either rank or bank level
112  void inject_refresh(bool b_ref_rank) {
113  // Rank-level refresh
114  if (b_ref_rank) {
115  for (auto rank : ctrl->channel->children)
116  refresh_target(ctrl, rank->id, -1, -1);
117  }
118  // Bank-level refresh. Simultaneously issue to all ranks (better performance than staggered refreshes).
119  else {
120  for (auto rank : ctrl->channel->children)
121  refresh_target(ctrl, rank->id, bank_ref_counters[rank->id], -1);
122  }
123  refreshed = clk;
124  }
