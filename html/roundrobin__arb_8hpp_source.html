64  {

66  // offset, should a request at input1 with priority pri1 supersede a
67  // request at input2 with priority pri2?
68  return ((pri1 > pri2) ||
69  ((pri1 == pri2) &&
70  (((input1 - offset + size) % size) < ((input2 - offset + size) % size))));
