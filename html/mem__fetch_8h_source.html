118  void kain_transform_to_HBM_Cache_address()
119  {
120  new_addr_type kain_bank_addr = (get_addr() >> 7)& 0x000f;
121  new_addr_type kain_column_addr = (get_addr() >> 11) & 0x001f;
122  new_addr_type kain_row_addr = (get_addr() >> 21) & 0x7fff;
123 
124 
125  kain_HBM_cache_channel = (get_addr()>>16) & 0x7;
126 
127 
128  kain_row_addr = 0x3fff-(kain_row_addr%0x7ff);//each channel 128MB, 8 channels so 1GB per GPU
129 
130  kain_new_addr = (kain_column_addr<<7) | (kain_bank_addr<<12) | (kain_row_addr<<16);
131  }
