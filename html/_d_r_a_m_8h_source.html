351 template <typename T>
352 void DRAM<T>::update_timing(typename T::Command cmd, const int* addr, long clk)
353 {
354  // I am not a target node: I am merely one of its siblings
355  if (id != addr[int(level)]) {
356  for (auto& t : timing[int(cmd)]) {
357  if (!t.sibling)
358  continue; // not an applicable timing parameter
359 
360  assert (t.dist == 1);
361 
362  long future = clk + t.val;
363  next[int(t.cmd)] = max(next[int(t.cmd)], future); // update future
364  }
365 
366  return; // stop recursion: only target nodes should be recursed
367  }
368 
369  // I am a target node
370  if (prev[int(cmd)].size()) {
371  prev[int(cmd)].pop_back(); // FIXME TIANSHI why pop back?
372  prev[int(cmd)].push_front(clk); // update history
373  }
374 
375  for (auto& t : timing[int(cmd)]) {
376  if (t.sibling)
377  continue; // not an applicable timing parameter
378 
379  long past = prev[int(cmd)][t.dist - 1];
380  if (past < 0)
381  continue; // not enough history
382 
383  long future = past + t.val;
384  next[int(t.cmd)] = max(next[int(t.cmd)], future); // update future
385  // TIANSHI: for refresh statistics
386  if (spec->is_refreshing(cmd) && spec->is_opening(t.cmd)) {
387  assert(past == clk);
388  begin_of_refreshing = clk;
389  end_of_refreshing = max(end_of_refreshing, next[int(t.cmd)]);
390  refresh_cycles += end_of_refreshing - clk;
391  if (cur_serving_requests > 0) {
392  refresh_intervals.push_back(make_pair(begin_of_refreshing, end_of_refreshing));
393  }
394  }
395  }
396 
397  // Some commands have timings that are higher that their scope levels, thus
398  // we do not stop at the cmd's scope level
399  if (!children.size())
400  return; // stop recursion: updated all levels
401 
402  // recursively update *all* of my children
403  for (auto child : children)
404  child->update_timing(cmd, addr, clk);

408 template <typename T>
409 void DRAM<T>::update_serving_requests(const int* addr, int delta, long clk) {
410  assert(id == addr[int(level)]);
411  assert(delta == 1 || delta == -1);
412  // update total serving requests
413  if (begin_of_cur_reqcnt != -1 && cur_serving_requests > 0) {
414  serving_requests += (clk - begin_of_cur_reqcnt) * cur_serving_requests;
415  active_cycles += clk - begin_of_cur_reqcnt;
416  }
417  // update begin of current request number
418  begin_of_cur_reqcnt = clk;
419  cur_serving_requests += delta;
420  assert(cur_serving_requests >= 0);
421 
422  if (delta == 1 && cur_serving_requests == 1) {
423  // transform from inactive to active
424  begin_of_serving = clk;
425  if (end_of_refreshing > begin_of_serving) {
426  active_refresh_overlap_cycles += end_of_refreshing - begin_of_serving;
427  }
428  } else if (cur_serving_requests == 0) {
429  // transform from active to inactive
430  assert(begin_of_serving != -1);
431  assert(delta == -1);
432  active_cycles += clk - begin_of_cur_reqcnt;
433  end_of_serving = clk;
434 
435  for (const auto& ref : refresh_intervals) {
436  active_refresh_overlap_cycles += min(end_of_serving, ref.second) - ref.first;
437  }
438  refresh_intervals.clear();
439  }
440 
441  int child_id = addr[int(level) + 1];
442  // We only count the level bank or the level higher than bank
443  if (child_id < 0 || !children.size() || (int(level) > int(T::Level::Bank)) ) {
444  return;
445  }
446  children[child_id]->update_serving_requests(addr, delta, clk);
447 }
