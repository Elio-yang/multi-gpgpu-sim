170 class functionalCoreSim: public core_t
171 {
172 public:
173  functionalCoreSim(kernel_info_t * kernel, gpgpu_sim *g, unsigned warp_size)
174  : core_t( g, kernel, warp_size, kernel->threads_per_cta() )
175  {
176  m_warpAtBarrier = new bool [m_warp_count];
177  m_liveThreadCount = new unsigned [m_warp_count];
178  }
179  virtual ~functionalCoreSim(){
180  warp_exit(0);
181  delete[] m_liveThreadCount;
182  delete[] m_warpAtBarrier;
183  }
185  void execute();
186  virtual void warp_exit( unsigned warp_id );
187  virtual bool warp_waiting_at_barrier( unsigned warp_id ) const
188  {
189  return (m_warpAtBarrier[warp_id] || !(m_liveThreadCount[warp_id]>0));
190  }
191 
192 private:
193  void executeWarp(unsigned, bool &, bool &);
194  //initializes threads in the CTA block which we are executing
195  void initializeCTA();
196  virtual void checkExecutionStatusAndUpdate(warp_inst_t &inst, unsigned t, unsigned tid)
197  {
198  if(m_thread[tid]==NULL || m_thread[tid]->is_done()){
199  m_liveThreadCount[tid/m_warp_size]--;
200  }
201  }
202 
203  // lunches the stack and set the threads count
204  void createWarp(unsigned warpId);
205 
206  //each warp live thread count and barrier indicator
207  unsigned * m_liveThreadCount;
208  bool* m_warpAtBarrier;
209 };
